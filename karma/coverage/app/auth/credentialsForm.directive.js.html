/**
 * @ngdoc directive
 * @name feCredentialsForm
 * @description
 * Form for login or signup
 */
 
 
(function () {
  'use strict';
 
  angular
    .module('app.Auth')
    .directive('feCredentialsForm', directiveFunc);
 
  /** @ngInject */
  function directiveFunc() {
    var directive = {
      restrict: 'E',
      templateUrl: 'app/auth/credentialsForm.html',
      scope: {
        errors: '=',
        submit: '&',
        clearErrors: '&',
        entity: '=',
        ok: '@'
      },
      controller: Controller,
      controllerAs: 'vm',
      bindToController: true
    };
 
    return directive;
  }
 
  /** @ngInject */
  function Controller() {
    var vm = this;
 
    activate();
 
    function activate() {
      vm.clearUsernameErrors = function() {
        vm.clearErrors(vm.errors, ['username', 'other'])
      };
      vm.clearPasswordErrors = function() {
        vm.clearErrors(vm.errors, ['password', 'other'])
      }
     }
  }
})();

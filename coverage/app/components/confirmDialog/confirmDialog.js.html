(function () {
  'use strict';
 
 
  var confirmModalDefaults = {
    template: '<div class="modal-header"><h3 class="modal-title">{{vm.data.title}}</h3></div>' +
    '<div class="modal-body">{{vm.data.text}}</div>' +
    '<div class="modal-footer">' +
    '<button class="btn btn-primary" ng-click="vm.ok()">{{vm.data.ok}}</button>' +
    '<button class="btn btn-default" ng-click="vm.cancel()">{{vm.data.cancel}}</button>' +
    '</div>',
    controller: 'ConfirmModalController',
    controllerAs: 'vm',
    defaultLabels: {
      title: 'Confirm',
      ok: 'OK',
      cancel: 'Cancel'
    }
 
  };
 
 
  angular
    .module('frontend')
    .value('$confirmModalDefaults', confirmModalDefaults)
    .controller('ConfirmModalController', confirmModalController)
    .factory('$confirm', confirm);
 
 
  /** @ngInject */
  function confirmModalController($uibModalInstance, data) {
    var vm = this;
    vm.data = angular.copy(data);
 
    vm.ok = function (closeMessage) {
      $uibModalInstance.close(closeMessage);
    };
 
    vm.cancel = function (dismissMessage) {
      if (angular.isUndefined(dismissMessage)) {
        dismissMessage = 'cancel';
      }
      $uibModalInstance.dismiss(dismissMessage);
    };
 
  }
 
 
  /** @ngInject */
  function confirm($uibModal, $confirmModalDefaults) {
    return function (data, settings) {
      var defaults = angular.copy($confirmModalDefaults);
      settings = angular.extend(defaults, (settings || {}));
 
      data = angular.extend({}, settings.defaultLabels, data || {});
 
      if ('templateUrl' in settings && 'template' in settings) {
        delete settings.template;
      }
 
      settings.resolve = {
        data: function () {
          return data;
        }
      };
 
      return $uibModal.open(settings).result;
    };
  }
 
})();

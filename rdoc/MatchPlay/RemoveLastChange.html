def remove_last
  @save_list = []
  @destroy_list = []
  if match.team_winner
    remove_match_complete
  elsif match.last_set
    remove_last_set_change(match.last_set)
  end
  save_list.each(&:save!)
  destroy_list.each(&:destroy)
end

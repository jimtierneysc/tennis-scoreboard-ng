def near_winners
  result = {
    set: [],
    match: []
  }
  first = object.first_team
  second = object.second_team
  last_set_var = object.last_set
  if first && second && last_set_var
    add_near_winner first, last_set_var, result
    add_near_winner second, last_set_var, result
  end

  result
 end
